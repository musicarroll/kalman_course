\begin{frame}
  \onslide<1->{\frametitle{Overview of the Examples}}
\begin{itemize}
  \onslide<2->{\item Discrete-time systems}
  \onslide<3->{\item Gradually increasing complexity:  }
  \begin{itemize}
    \onslide<4->{\item Resistor: 1 state, \textbf{static} (constant dynamics)  }
    \onslide<5->{\item Constant Velocity Motion:  2 states, with one state \textbf{dynamic} and one \textbf{static} }
    \onslide<6->{\item Damped Harmonic Oscillator:  2 states, with \textbf{both states dynamic}  }
  \end{itemize}
\end{itemize}
\onslide<7->{\center --------------------------------------------------------------------}
\end{frame}
