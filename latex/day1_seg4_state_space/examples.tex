\begin{frame}
  \onslide<1->{\frametitle{Vector Dynamics: Damped Harmonic Oscillator}
}
 \begin{itemize}
 \onslide<2->{\item MATLAB/Simulink Model
}
 \onslide<3->{\item 2-dimensional Vector Dynamics
}
 \onslide<4->{\item Deterministic, Continuous Model
}
\onslide<5->{
\[
\begin{bmatrix}
\dot{x}_1\\
\dot{x}_2
\end{bmatrix} =
\begin{bmatrix}
0 & 1\\
-k/m & -c/m
\end{bmatrix}
\begin{bmatrix}
x_1\\
x_2
\end{bmatrix}
\]
}
\end{itemize}

\end{frame}

\begin{frame}
  \onslide<1->{\frametitle{Vector Dynamics: Damped Harmonic Oscillator}
}
 \begin{itemize}
 \onslide<2->{\item Discrete model would look like this}
\onslide<3->{ 
\[\begin{bmatrix}
x_{1}(k)\\
x_{2}(k)
\end{bmatrix}=\exp
\left(
\begin{bmatrix}
0 & 1\\
-k/m & -c/m
\end{bmatrix}
\Delta t
\right)
\begin{bmatrix}
x_{1}(k-1)\\
x_{2}(k-1)
\end{bmatrix}
\]
}
\onslide<3->{\item Here $\Delta t=t_k - t_{k-1}$
}
\onslide<4->{\item Note that $\exp$ is a matrix exponential
}
\onslide<5->{\item In the continuous case, the state transition matrix is a matrix exponential whenever the system dynamics matrix is a constant (i.e., time invariant) matrix
}
\end{itemize}
\onslide<6->{\center --------------------------------------------------------------------}
\end{frame}
